```sql
# COALESCE 사용 풀이
SELECT WAREHOUSE_ID, WAREHOUSE_NAME, ADDRESS, COALESCE(FREEZER_YN, 'N')
FROM FOOD_WAREHOUSE
WHERE ADDRESS like '경기도%';

# IFNULL 사용 풀이
SELECT WAREHOUSE_ID, WAREHOUSE_NAME, ADDRESS, IFNULL(FREEZER_YN, 'N') AS FREEZER_YN 
FROM FOOD_WAREHOUSE 
WHERE ADDRESS 
LIKE '%경기도%' 
ORDER BY WAREHOUSE_ID ASC;
```
